Custom DTO Object

ItemChat.java

import com.google.firebase.Timestamp;
import com.google.firebase.firestore.Exclude;
import java.util.Date;

public class ItemChat implements Comparable {
    private String name;
    private String content;
    private Timestamp timestamp;
    // Firestore에서 수신 시에만 Document ID 저장용으로 사용.
    // Firestore에 저장 시에는 사용되지 않는 필드이므로 Exclude.
    @Exclude
    private String id;

    public ItemChat() {
        this("", "", new Timestamp(new Date()));
    }

    public ItemChat(String name, String content, Timestamp timestamp) {
        this.name = name;
        this.content = content;
        this.timestamp = timestamp;
    }

    public Timestamp getTimestamp() {
        return timestamp;
    }

    @Exclude
    public String getId() {
        return id;
    }

    public String getContent() {
        return content;
    }

    public String getName() {
        return name;
    }

    public void setId(String id) {
        this.id = id;
    }

    public void setContent(String content) {
        this.content = content;
    }

    public void setName(String name) {
        this.name = name;
    }

    public void setTimestamp(Timestamp timestamp) {
        this.timestamp = timestamp;
    }

    // Timestamp로 Sort시 사용.
    @Override
    public int compareTo(Object o) {
        return this.timestamp.compareTo(((ItemChat) o).timestamp);
    }
}

RecyclerView Adapter

ChatAdapter.java

public class ChatAdapter extends RecyclerView.Adapter<ChatAdapter.ViewHolder> {
    private ArrayList<ItemChat> mData = new ArrayList<>();

    @NonNull
    @Override
    public ViewHolder onCreateViewHolder(@NonNull ViewGroup parent, int viewType) {
        Context context = parent.getContext();
        LayoutInflater inflater = (LayoutInflater) context.getSystemService(Context.LAYOUT_INFLATER_SERVICE);
        View view = Objects.requireNonNull(inflater).inflate(R.layout.item_chat, parent, false);
        return new ViewHolder(view);
    }

    @Override
    public void onBindViewHolder(@NonNull ViewHolder holder, int position) {
        ItemChat item = mData.get(position);
        holder.tvName.setText(item.getName());
        holder.tvContent.setText(item.getContent());
        String timestamp = new SimpleDateFormat("yyyy-MM-dd HH:mm:ss", Locale.KOREA).format(item.getTimestamp().toDate());
        holder.tvTime.setText(timestamp);
    }

    @Override
    public int getItemCount() {
        return mData.size();
    }

    public void setData(ArrayList<ItemChat> data) {
        mData.clear();
        mData.addAll(data);
        Collections.sort(mData);
        notifyDataSetChanged();
    }

    class ViewHolder extends RecyclerView.ViewHolder {
        TextView tvName, tvContent, tvTime;

        ViewHolder(@NonNull View itemView) {
            super(itemView);

            tvName = itemView.findViewById(R.id.item_chat_name);
            tvContent = itemView.findViewById(R.id.item_chat_content);
            tvTime = itemView.findViewById(R.id.item_chat_timestamp);
        }
    }
}

Activity

MainActivity.java

Global Variables
private static final String TAG = MainActivity.class.getSimpleName();
private ChatAdapter mAdapter;
private FirebaseFirestore mDatabase;
private ListenerRegistration mRegistration;
private ArrayList<ItemChat> mData;
private EditText etName, etContent;
OnCreate
@Override
protected void onCreate(Bundle savedInstanceState) {
	super.onCreate(savedInstanceState);
	setContentView(R.layout.activity_main);

	RecyclerView recyclerView = findViewById(R.id.main_list);
	recyclerView.setLayoutManager(new LinearLayoutManager(this));
	mAdapter = new ChatAdapter();
	recyclerView.setAdapter(mAdapter);

	mDatabase = FirebaseFirestore.getInstance();
	mData = new ArrayList<>();
	etName = findViewById(R.id.main_et_name);
	etContent = findViewById(R.id.main_et_content);
	Button btn = findViewById(R.id.main_btn_send);
	btn.setOnClickListener(new View.OnClickListener() {
		@Override
		public void onClick(View v) {
			String name = etName.getText().toString();
			String content = etContent.getText().toString();

			if (!TextUtils.isEmpty(name) && !TextUtils.isEmpty(content)) {
				ItemChat item = new ItemChat(name, content, new Timestamp(new Date()));
				mDatabase.collection("chats").add(item).addOnCompleteListener(new OnCompleteListener<DocumentReference>() {
					@Override
					public void onComplete(@NonNull Task<DocumentReference> task) {
						if (task.isSuccessful()) {
							etName.setText("");
							etContent.setText("");
						} else {
							Toast.makeText(MainActivity.this, "전송에 실패했습니다.", Toast.LENGTH_SHORT).show();
						}
					}
				});
			}
		}
	});
}
OnStart
@Override
protected void onStart() {
	super.onStart();

	mRegistration = mDatabase.collection("chats").addSnapshotListener(new EventListener<QuerySnapshot>() {
		@Override
		public void onEvent(@Nullable QuerySnapshot snapshots, @Nullable FirebaseFirestoreException e) {
			if (e != null) {
				Log.w(TAG, "Failed", e);
				return;
			}

			for (DocumentChange dc : snapshots.getDocumentChanges()) {
				switch (dc.getType()) {
					case ADDED:
						ItemChat item = dc.getDocument().toObject(ItemChat.class);
						item.setId(dc.getDocument().getId());
						mData.add(item);
						mAdapter.setData(mData);
						break;
					case MODIFIED:
						// Document Modified.
						break;
					case REMOVED:
						String _id = dc.getDocument().getId();
						int idx = -1;
						for (int i = 0; i < mData.size(); i++) {
							if (mData.get(i).getId().equals(_id)) {
								idx = i;
								break;
							}
						}
						if (idx >= 0) {
							mData.remove(idx);
							mAdapter.setData(mData);
						}
						break;
				}
			}
		}
	});
}
OnStop
@Override
protected void onStop() {
	if (mRegistration != null) mRegistration.remove();
	super.onStop();
}
